{
  "v": "2",
  "ps": "<remarks>",
  "add": "<address>",
  "port": "<port>",
  "id": "<id>",
  "aid": "<alterId>",
  "scy": "<cipher>",
  "net": "<tcp, kcp, ws, httpupgrade, xhttp, h2, quic, grpc>",

  "path": "<path>",

  "sni": "<sni>",
  "alpn": "<alpn>",
  "fp": "<chrome, firefox, safari, ios, android, edge, 360, qq, random, randomized, none>"
}
